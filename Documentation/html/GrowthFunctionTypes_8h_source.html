275  GridBasedGrowthFunction(int id, int type, float initTime, float endTime, bool applyToColumnarLayer, bool applyToPeripodialMembrane, int nX, int nY, double*** GrowthMat, double** AngleMat) : GrowthFunctionBase(id, type, initTime, endTime, applyToColumnarLayer, applyToPeripodialMembrane){
282  this ->nGridX = nX;
283  this ->nGridY = nY;
284  GrowthMatrix = new double**[(const int) nGridX];
285  xyShearAngleMatrix = new double*[(const int) nGridX];
286  for (int i=0; i<nGridX; ++i){
287  GrowthMatrix[i] = new double*[(const int) nGridY];
288  xyShearAngleMatrix[i] = new double[(const int) nGridY];
289  for (int j=0; j<nGridY; ++j){
290  GrowthMatrix[i][j] = new double[3];
291  xyShearAngleMatrix[i][j] = AngleMat[i][j]*M_PI/180.0; //converting to radians
292  for (int k=0; k<3; ++k){
293  GrowthMatrix[i][j][k] = GrowthMat[i][j][k];
294  }
295  }
296  }
297 
298  xyShearRotationsMatrix = new gsl_matrix**[(const int) nGridX];
299  for (int i=0; i<nGridX; ++i){
300  xyShearRotationsMatrix[i] = new gsl_matrix*[(const int) nGridY];
301  for (int j=0; j<nGridY; ++j){
302  xyShearRotationsMatrix[i][j] = new gsl_matrix;
303  xyShearRotationsMatrix[i][j] = gsl_matrix_calloc(3,3);
304  double c = cos(xyShearAngleMatrix[i][j]);
305  double s = sin(xyShearAngleMatrix[i][j]);
306  //cout<<"xyShearRotationsMatrix ["<<i<<"]["<<j<<"], angle: "<<xyShearAngleMatrix[i][j]<<" cos: "<<c<<" sin "<<s<<endl;
307  gsl_matrix_set(xyShearRotationsMatrix[i][j],0,0, c );
308  gsl_matrix_set(xyShearRotationsMatrix[i][j],0,1, -1.0*s);
309  gsl_matrix_set(xyShearRotationsMatrix[i][j],0,2, 0.0);
310  gsl_matrix_set(xyShearRotationsMatrix[i][j],1,0, s);
311  gsl_matrix_set(xyShearRotationsMatrix[i][j],1,1, c);
312  gsl_matrix_set(xyShearRotationsMatrix[i][j],1,2, 0.0);
313  gsl_matrix_set(xyShearRotationsMatrix[i][j],2,0, 0.0);
314  gsl_matrix_set(xyShearRotationsMatrix[i][j],2,1, 0.0);
315  gsl_matrix_set(xyShearRotationsMatrix[i][j],2,2, 1.0);
316  }
317  }
318  }
